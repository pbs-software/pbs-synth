\onefig{traceParams}{MCMC traces for the estimated parameters. Grey lines show the \Nmcmc~samples for each parameter, solid lines show the cumulative median (up to that sample), and dashed lines show the cumulative 0.05 and 0.95 quantiles.  Red circles are the MPD estimates. For parameters other than $M$ (if estimated), subscripts 1-7 correspond to fleets (fisheries and surveys).}{Base run: }{ytr.}
\onefig{splitChain}{Diagnostic plot obtained by dividing the MCMC chain of \Nmcmc~MCMC samples into eight segments (original number of chains), and overplotting the cumulative distributions of the segments from first to eighth using colours pink, brown, purple, orange, green, black, blue, and red.}{Base run: }{ytr.}
\onefig{paramACFs}{Autocorrelation plots for the estimated parameters from the MCMC output. Horizontal dashed blue lines delimit the 95\pc{} confidence interval for each parameter's set of lagged correlations.}{Base run: }{ytr.}
\clearpage

\twofig{rhat}{rhat.recdev}{Scale-reduction split-$\widehat{R}$ statistic measures the ratio of the average variance of draws within each chain to the variance of the pooled draws across chains: parameters (top), recruitment deviations (bottom). Pale green bars show effective sample size (ESS). Vertical dashed red line at $\widehat{R}$=1.01 marks acceptable convergence boundary; vertical dashed purple line marks ESS=400. When $\widehat{R}$>1.01 or ESS<400, parameter estimation has not converged.}{Base run: }{ytr.}
\onefig{rhat.hist.mcmc}{Stacked histogram of posteriors from eight chains for female $M$ (top left), male $M$ (top right), $\log R_0$ (bottom left), and steepness $h$ (bottom right).}{Base run: }{ytr.}
